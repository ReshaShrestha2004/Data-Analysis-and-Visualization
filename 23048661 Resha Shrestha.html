Statistical Testing¶

Test 1: Whether the average response time across complaint types is similar or not.¶

1.1) Stating Null Hypothesis¶
Null Hypothesis (H0): The average response time across complaint types is similar.¶
Alternate Hypothesis (H1): At least one complaint type has a different average response time.¶
1.2) Performing the statistical test¶
In [177]:
import scipy.stats as stats
# Getting the top 5 most frequent complaint types from the dataset
Top_5_complaints = df4['Complaint Type'].value_counts().head(5).index.tolist()
# Filtering and making a dataframe to include only the top 5 complaint types
filtered_df = df4[df4['Complaint Type'].isin(Top_5_complaints)]
# Initializing empty lists to store response time data for each complaint type
complaint_groups = []  
complaint_names = []  
for complaint in Top_5_complaints:
   # Extract response hours for the current complaint type of the iteration
   group_data = filtered_df[filtered_df['Complaint Type'] == complaint]['Response_Hours'].values
   # Only include complaint types that have data
   if len(group_data) > 0:  
       complaint_groups.append(group_data)  # Add response times array to list
       complaint_names.append(complaint)    # Add complaint name to list
# Calculating the average response time for each complaint type
complaint_averages = [group.mean() for group in complaint_groups]
# Create a pandas Series with complaint types as index and average times as values 
complaint_times = pd.Series(complaint_averages, index=complaint_names)
print("-" * 120)
print("Analysis of Average Response Time by Top 5 Complaint Types")
print("-" * 120)
# Printing the average response time for each complaint type
print("Top 5 Complaint Types and their average response time:")
for complaint, avg_time in zip(complaint_names, complaint_averages):
   print(f"{complaint}: {avg_time:.2f} hours")
# Performing one-way ANOVA test to determine if the differences in response times are statistically significant
f_statistic, p_value = stats.f_oneway(*complaint_groups)
print(f"\n\n")
print("-" * 120)
print("Statistical Testing:")
print("-" * 120)
# Print the F-statistic 
print(f"ANOVA F-statistic: {f_statistic:.2f}")
# Print the p-value
print(f"p-value: {p_value:.4f}")
print("_" * 120)

# Creating a pie chart visualization to show the proportion of response times for each complaint type
plt.figure(figsize=(10, 8))
# Create pie chart with percentages shown on each slice and labels for complaint type
plt.pie(complaint_times, labels=complaint_times.index, autopct='%1.1f%%')
plt.title('Distribution of Average Response Time by Top 5 Complaint Types')
# Create legend labels at right that include both the complaint type and average hours
legend_labels = [f"{complaint}: {hours:.1f} hrs" for complaint, hours in zip(complaint_times.index, complaint_times.values)]
plt.legend(legend_labels, loc='best', bbox_to_anchor=(1, 0.5))
plt.tight_layout()
plt.show()
